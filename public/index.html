wget -qO- https://dl.packager.io/srv/opf/openproject/key | sudo apt-key add - [This changes to the next line because apt-key is deprecated] wget -qO- https://dl.packager.io/srv/opf/openproject/key | sudo tee /etc/apt/trusted.gpg.d/ sudo wget -O /etc/apt/sources.list.d/openproject.list
https://dl.packager.io/srv/opf/openproject/release/15.1/installer/ubuntu/22.04.repo sudo apt-get update sudo apt-get install openproject
